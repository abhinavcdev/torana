# caddy.rs v0.1 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a minimal Rust-based reverse proxy with HTTP/1.1, HTTP/2, TLS termination, round-robin load balancing, structured logging, Prometheus metrics, and zero-downtime SIGHUP config reload.

**Architecture:** 
- Single-process async proxy using tokio runtime
- TOML configuration with atomic reload via channels
- Pluggable listener stack (HTTP, HTTPS with rustls)
- Connection pooling and per-upstream load balancing
- Structured access logging (JSON) and Prometheus metrics on dedicated endpoint
- Graceful shutdown: drain in-flight requests on SIGTERM

**Tech Stack:** Tokio, Hyper 1.x, Rustls, Serde (TOML), Prometheus client, Tracing (structured logging), Clap (CLI)

```rust
// src/upstream.rs
use std::sync::Arc;
use std::collections::HashMap;
use hyper::Client;
use tokio::sync::RwLock;

#[derive(Clone)]
pub struct UpstreamPool {
    pub addr: String,
    pub weight: u32,
}

pub struct LoadBalancer {
    upstreams: Vec<UpstreamPool>,
    counter: Arc<RwLock<usize>>,
}

impl LoadBalancer {
    pub fn new(upstreams: Vec<UpstreamPool>) -> Self {
        Self {
            upstreams,
            counter: Arc::new(RwLock::new(0)),
        }
    }
    
    pub async fn next(&self) -> Option<&UpstreamPool> {
        let mut counter = self.counter.write().await;
        if self.upstreams.is_empty() {
            return None;
        }
        *counter = (*counter + 1) % self.upstreams.len();
        Some(&self.upstreams[*counter])
    }
}
```

**Step 2: Create proxy request handler**

```rust
// src/proxy.rs
use hyper::{Client, Request, Response, Body};
use std::net::SocketAddr;
use tokio::time::Instant;

pub async fn proxy_request(
    req: Request<Body>,
    upstream: &str,
) -> anyhow::Result<Response<Body>> {
    let start = Instant::now();
    let client = Client::new();
    
    // Reconstruct URI for upstream
    let original_path = req.uri().path_and_query()
        .map(|pq| pq.as_str())
        .unwrap_or("/");
    
    let upstream_uri = format!("{}{}", upstream, original_path);
    
    let mut upstream_req = Request::builder()
        .method(req.method().clone())
        .uri(&upstream_uri);
    
    // Copy headers
    for (key, value) in req.headers().iter() {
        upstream_req = upstream_req.header(key, value.clone());
    }
    
    let body = hyper::body::to_bytes(req.into_body()).await?;
    let upstream_req = upstream_req.body(Body::from(body))?;
    
    let response = client.request(upstream_req).await?;
    let latency_ms = start.elapsed().as_millis() as u64;
    
    tracing::info!(
        status = response.status().as_u16(),
        latency_ms = latency_ms,
        upstream = upstream,
        "Proxied request"
    );
    
    Ok(response)
}
```

```rust
    metrics: Metrics,

async fn handle_request(
    req: Request<Body>,
    routes: Vec<config::RouteConfig>,
    metrics: Metrics,
) -> Result<Response<Body>, Box<dyn std::error::Error>> {
    metrics.http_requests_total.inc();
    
    // First route always used (routing logic in Task 8)
    if let Some(route) = routes.first() {
        if let Some(upstream_cfg) = route.upstream.first() {
            match proxy_request(req, &upstream_cfg.addr).await {
                Ok(resp) => {
                    return Ok(resp);
                }
                Err(e) => {
                    tracing::error!("Upstream error: {}", e);
                    metrics.upstream_connection_errors.inc();
                    return Ok(Response::builder()
                        .status(502)
                        .body(Body::from("Bad Gateway"))
                        .unwrap());
                }
            }
        }
    }
    
    Ok(Response::builder()
        .status(404)
        .body(Body::from("Not Found"))
        .unwrap())
}
```

```rust
// src/tls.rs
use rustls_pemfile::{certs, private_keys};
use std::fs::File;
use std::io::BufReader;
use tokio_rustls::TlsAcceptor;

pub fn load_tls_config(cert_path: &str, key_path: &str) -> anyhow::Result<TlsAcceptor> {
    let cert_file = BufReader::new(File::open(cert_path)?);
    let key_file = BufReader::new(File::open(key_path)?);
    
    let mut certs_vec = certs(cert_file).collect::<Result<Vec<_>, _>>()?;
    let mut keys = private_keys(key_file).collect::<Result<Vec<_>, _>>()?;
    
    if certs_vec.is_empty() {
        anyhow::bail!("No certificates found");
    }
    if keys.is_empty() {
        anyhow::bail!("No private keys found");
    }
    
    let config = rustls::ServerConfig::builder()
        .with_no_client_auth()
        .with_single_cert(certs_vec, keys.remove(0))?;
    
    Ok(TlsAcceptor::from(std::sync::Arc::new(config)))
}
```

```rust
}
```

```rust
// src/reload.rs
use std::sync::Arc;
use tokio::sync::RwLock;

pub type ConfigHandle = Arc<RwLock<Config>>;

pub async fn watch_config_signal(config_path: String, config_handle: ConfigHandle) {
    use tokio::signal::unix::{signal, SignalKind};
    
    let mut sighup = signal(SignalKind::hangup())
        .expect("Failed to install SIGHUP handler");
    
    loop {
        sighup.recv().await;
        tracing::info!("SIGHUP received, reloading config");
        
        match config::load_config(&config_path) {
            Ok(new_config) => {
                let mut handle = config_handle.write().await;
                *handle = new_config;
                tracing::info!("Config reloaded successfully");
            }
            Err(e) => {
                tracing::error!("Failed to reload config: {}", e);
                // Keep old config on error
            }
        }
    }
}
```

```rust
// src/upstream.rs - update LoadBalancer:

impl LoadBalancer {
    pub fn new(upstreams: Vec<UpstreamPool>) -> Self {
        // Expand based on weights for simple round-robin
        let expanded = upstreams.iter().flat_map(|u| {
            let weight = u.weight.unwrap_or(100);
            (0..weight).map(move |_| u.clone())
        }).collect();
        
        Self {
            upstreams: expanded,
            counter: Arc::new(RwLock::new(0)),
        }
    }
    
    pub async fn next(&self) -> Option<&UpstreamPool> {
        if self.upstreams.is_empty() {
            return None;
        }
        let mut counter = self.counter.write().await;
        let idx = *counter % self.upstreams.len();
        *counter = (*counter + 1) % self.upstreams.len();
        Some(&self.upstreams[idx])
    }
}
```

**Step 2: Update request handler to use load balancer**

```rust
// In main.rs, update handle_request:

async fn handle_request(
    req: Request<Body>,
    routes: Vec<config::RouteConfig>,
    metrics: Metrics,
) -> Result<Response<Body>, Box<dyn std::error::Error>> {
    metrics.http_requests_total.inc();
    
    if let Some(route) = routes.first() {
        let lb = upstream::LoadBalancer::new(
            route.upstream.iter().map(|u| upstream::UpstreamPool {
                addr: u.addr.clone(),
                weight: u.weight,
            }).collect()
        );
        
        if let Some(upstream) = lb.next().await {
            match proxy_request(req, &upstream.addr).await {
                Ok(resp) => return Ok(resp),
                Err(e) => {
                    tracing::error!("Upstream error: {}", e);
                    metrics.upstream_connection_errors.inc();
                }
            }
        }
    }
    
    Ok(Response::builder()
        .status(502)
        .body(Body::from("Bad Gateway"))
        .unwrap())
}
```
